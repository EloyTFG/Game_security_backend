exports.submitComment = (req, res) => {
    const { comment } = req.body;

        explicitArray: false,
        ignoreAttrs: false,
        normalizeTags: true,
        explicitRoot: true,
        dtd: true 
    });
 
    parser.parseString(comment, (err, result) => {
        if (err) {
            return res.status(400).json({ message: 'Error al procesar el comentario', error: err.message });
